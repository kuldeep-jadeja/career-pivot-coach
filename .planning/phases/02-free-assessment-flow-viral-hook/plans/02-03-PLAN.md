```typescript
'use client';

import { useRouter } from 'next/navigation';
import { Button } from '@/app/_components/ui/button';
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from '@/app/_components/ui/card';
import { IndustrySelect } from '@/app/_components/assessment/IndustrySelect';
import { ExperienceInput } from '@/app/_components/assessment/ExperienceInput';
import { ProgressTracker } from '@/app/_components/assessment/ProgressTracker';
import { useAssessmentFlow } from '@/lib/hooks/useAssessmentFlow';
import { useState } from 'react';
import { calculateAndSaveAssessment } from '@/app/_actions/assessment';
import { toast } from 'sonner';

export default function DetailsPage() {
  const router = useRouter();
  const { assessment, setField, isHydrated } = useAssessmentFlow();
  const [isCalculating, setIsCalculating] = useState(false);

  const handleSubmit = async () => {
    if (!assessment.jobTitle?.code) {
      router.push('/start');
      return;
    }

    setIsCalculating(true);
    try {
      const result = await calculateAndSaveAssessment({
        jobTitle: assessment.jobTitle,
        industry: assessment.industry || 'Other',
        yearsExperience: assessment.yearsExperience ?? 0,
      });
      
      if (result.success) {
        router.push(`/results?id=${result.assessmentId}`);
      }
    } catch (error) {
      toast.error('Failed to calculate risk score. Please try again.');
      console.error(error);
    } finally {
      setIsCalculating(false);
    }
  };

  if (!isHydrated) {
    return <div className="h-screen flex items-center justify-center">Loading...</div>;
  }

  return (
    <div className="space-y-8">
      <ProgressTracker currentStep={2} totalSteps={3} />
      
      <Card>
        <CardHeader>
          <CardTitle className="text-2xl">Tell us a bit more</CardTitle>
          <CardDescription>
            This helps us provide a more accurate risk assessment.
          </CardDescription>
        </CardHeader>
        <CardContent className="space-y-6">
          <div className="space-y-4">
            <IndustrySelect
              value={assessment.industry || null}
              onChange={(value) => setField('industry', value)}
            />
            
            <ExperienceInput
              value={assessment.yearsExperience ?? 0}
              onChange={(value) => setField('yearsExperience', value)}
            />
          </div>
          
          <div className="flex justify-between pt-4">
            <Button 
              variant="outline"
              onClick={() => router.push('/start')}
              className="min-h-[44px]"
            >
              Back
            </Button>
            <Button 
              onClick={handleSubmit}
              disabled={isCalculating}
              className="min-h-[44px] px-8"
            >
              {isCalculating ? 'Calculating...' : 'See My Results'}
            </Button>
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
```

Create app/_actions/assessment.ts:
```typescript
'use server';

import { z } from 'zod';
import { calculateRiskScore } from '@/lib/scoring/risk-calculator';
import { createClient } from '@/lib/db/supabase';
import { nanoid } from 'nanoid';

const assessmentInput = z.object({
  jobTitle: z.object({
    code: z.string(),
    title: z.string(),
  }),
  industry: z.string(),
  yearsExperience: z.number().min(0).max(50),
});

export async function calculateAndSaveAssessment(
  input: z.infer<typeof assessmentInput>
) {
  const validated = assessmentInput.parse(input);
  
  // Calculate risk score
  const result = await calculateRiskScore({
    occupationCode: validated.jobTitle.code,
    industryCode: validated.industry,
    yearsExperience: validated.yearsExperience,
  });
  
  // Generate assessment ID
  const assessmentId = nanoid(12);
  
  // Save to database (anonymous - no userId)
  const supabase = createClient();
  const { error } = await supabase.from('assessments').insert({
    id: assessmentId,
    job_title: validated.jobTitle.title,
    occupation_code: validated.jobTitle.code,
    industry: validated.industry,
    years_experience: validated.yearsExperience,
    risk_score: result.overall,
    layer_breakdown: result.breakdown,
    risk_band: result.band.level,
    confidence: result.confidence,
    created_at: new Date().toISOString(),
  });

  if (error) {
    console.error('Failed to save assessment:', error);
    // Still return result even if save fails
  }

  return {
    success: true,
    assessmentId,
    riskScore: result.overall,
    breakdown: result.breakdown,
    band: result.band,
  };
}
```

Create app/(assessment)/results/page.tsx:
```typescript
import { Suspense } from 'react';
import { notFound } from 'next/navigation';
import { createClient } from '@/lib/db/supabase';
import { RiskScoreGauge } from '@/app/_components/results/RiskScoreGauge';
import { LayerBreakdown } from '@/app/_components/results/LayerBreakdown';
import { TaskRiskList } from '@/app/_components/results/TaskRiskList';
import { ScoreInterpretation } from '@/app/_components/results/ScoreInterpretation';
import { ProgressTracker } from '@/app/_components/assessment/ProgressTracker';
import { Card, CardContent, CardHeader, CardTitle } from '@/app/_components/ui/card';

interface ResultsPageProps {
  searchParams: Promise<{ id?: string }>;
}

async function getAssessment(id: string) {
  const supabase = createClient();
  const { data, error } = await supabase
    .from('assessments')
    .select('*')
    .eq('id', id)
    .single();
  
  if (error || !data) return null;
  return data;
}

export default async function ResultsPage({ searchParams }: ResultsPageProps) {
  const { id } = await searchParams;
  
  if (!id) {
    notFound();
  }

  const assessment = await getAssessment(id);
  
  if (!assessment) {
    notFound();
  }

  return (
    <div className="space-y-8">
      <ProgressTracker currentStep={3} totalSteps={3} />
      
      <Card>
        <CardHeader className="text-center">
          <CardTitle className="text-2xl">Your AI Displacement Risk</CardTitle>
        </CardHeader>
        <CardContent className="space-y-8">
          <RiskScoreGauge score={assessment.risk_score} />
          
          <ScoreInterpretation 
            score={assessment.risk_score}
            jobTitle={assessment.job_title}
          />
        </CardContent>
      </Card>

      <Card>
        <CardContent className="pt-6">
          <LayerBreakdown breakdown={assessment.layer_breakdown} />
        </CardContent>
      </Card>

      <Suspense fallback={<div>Loading tasks...</div>}>
        <Card>
          <CardContent className="pt-6">
            <TaskRiskList tasks={assessment.task_analysis || []} />
          </CardContent>
        </Card>
      </Suspense>
    </div>
  );
}
```
